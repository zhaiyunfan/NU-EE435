def sigmoid(z, d=False):
    """Compute the sigmoid function or its derivative."""
    sig = 1 / (1 + np.exp(-z))
    if d:
        return (
            sig * (1 - sig) + 1e-12
        )  # Add epsilon to avoid division by zero in gradients
    return sig


def relu(z, d=False):
    """Compute the ReLU function or its derivative."""
    if d:
        return (
            np.where(z > 0, 1, 0) + 1e-12
        )  # Use np.where for better handling with arrays
    return np.maximum(0, z)  # np.maximum is more idiomatic and clear

# Activation functions and their derivatives
def relu(x):
    return np.maximum(0, x)


def relu_derivative(x):
    return (x > 0).astype(x.dtype)


def softmax(x):
    exp_x = np.exp(x - np.max(x, axis=1, keepdims=True))
    return exp_x / np.sum(exp_x, axis=1, keepdims=True)


# Loss function
def compute_loss(y, y_pred):
    m = y.shape[0]
    log_likelihood = -np.log(y_pred[range(m), y])
    loss = np.mean(log_likelihood)
    return loss


# Batch normalization forward and backward
def batch_norm_forward(x, gamma, beta, eps=1e-8):
    mean = np.mean(x, axis=0)
    variance = np.var(x, axis=0)
    x_norm = (x - mean) / np.sqrt(variance + eps)
    out = gamma * x_norm + beta
    cache = (x, x_norm, mean, variance, gamma, beta)
    return out, cache


def batch_norm_backward(dout, cache):
    x, x_norm, mean, variance, gamma, beta = cache
    N = x.shape[0]
    x_mu = x - mean
    std_inv = 1.0 / np.sqrt(variance + 1e-8)

    dx_norm = dout * gamma
    dvar = np.sum(dx_norm * x_mu, axis=0) * -0.5 * std_inv**3
    dmu = np.sum(dx_norm * -std_inv, axis=0) + dvar * np.mean(-2.0 * x_mu, axis=0)

    dx = (dx_norm * std_inv) + (dvar * 2 * x_mu / N) + (dmu / N)
    dgamma = np.sum(dout * x_norm, axis=0)
    dbeta = np.sum(dout, axis=0)

    return dx, dgamma, dbeta

def initialize_parameters(layer_sizes):
    parameters = []
    for i in range(len(layer_sizes) - 1):
        W = np.random.randn(layer_sizes[i], layer_sizes[i + 1]) * np.sqrt(

        b = np.zeros((1, layer_sizes[i + 1]))
        gamma = np.ones((1, layer_sizes[i + 1]))
        beta = np.zeros((1, layer_sizes[i + 1]))
        parameters.append({"W": W, "b": b, "gamma": gamma, "beta": beta})
    return parameters


# Forward and backward passes
def forward_backward_propagation(X, y, parameters):
    caches = []
    activations = [X]
    A = X

    # Forward pass with batch normalization and ReLU
    for param in parameters[:-1]:  # Last layer without batch normalization
        Z = np.dot(A, param["W"]) + param["b"]
        BN, cache_bn = batch_norm_forward(Z, param["gamma"], param["beta"])
        A = relu(BN)
        activations.append(A)
        caches.append((param, cache_bn))

    # Output layer
    final_Z = np.dot(A, parameters[-1]["W"]) + parameters[-1]["b"]
    final_A = softmax(final_Z)
    activations.append(final_A)
    caches.append((parameters[-1], None))

    # Compute loss
    loss = compute_loss(y, final_A)

    # Backward pass
    grads = []
    dA = final_A
    dA[range(y.shape[0]), y] -= 1  # Derivative of cross-entropy loss with softmax
    dA /= y.shape[0]

    for i in reversed(range(len(caches))):
        param, cache_bn = caches[i]
        if cache_bn:
            dZ = relu_derivative(cache_bn[0]) * dA  # Apply relu derivative
            dA, dgamma, dbeta = batch_norm_backward(dZ, cache_bn)
            grads.append(
                {
                    "dW": np.dot(activations[i].T, dA),
                    "db": np.sum(dA, axis=0),
                    "dgamma": dgamma,
                    "dbeta": dbeta,
                }
            )
        else:
            dW = np.dot(activations[i].T, dA)
            db = np.sum(dA, axis=0)
            grads.append({"dW": dW, "db": db})
        dA = np.dot(dA, param["W"].T)

    grads.reverse()
    return loss, grads


# Update parameters
def update_parameters(parameters, grads, learning_rate):
    for param, grad in zip(parameters, grads):
        param["W"] -= learning_rate * grad["dW"]
        param["b"] -= learning_rate * grad["db"]
        if "dgamma" in grad and "dbeta" in grad:
            param["gamma"] -= learning_rate * grad["dgamma"]
            param["beta"] -= learning_rate * grad["dbeta"]

def initialize_weights(layers):

    for i in range(1, len(layers)):

            1.0 / layers[i - 1]
        )
        weights[f"b{i}"] = np.zeros((layers[i], 1))
    return weights


def tanh(x):
    return np.tanh(x)


def tanh_derivative(x):
    return 1 - np.tanh(x) ** 2


def forward_pass(X, weights, layers):
    activations = {"A0": X}
    for i in range(1, len(layers)):
        z = np.dot(weights[f"W{i}"], activations[f"A{i-1}"]) + weights[f"b{i}"]
        a = tanh(z)
        activations[f"Z{i}"] = z
        activations[f"A{i}"] = a
    return activations


def compute_loss(y, y_hat):
    return np.mean((y - y_hat) ** 2)


def backward_pass(activations, weights, layers, X, y):
    m = X.shape[1]
    y = y.reshape(1, -1)  # Reshape y to match output dimensions
    gradients = {}
    # Output layer error
    delta = (activations[f"A{len(layers)-1}"] - y) * tanh_derivative(
        activations[f"Z{len(layers)-1}"]
    )
    gradients[f"W{len(layers)-1}"] = (
        1 / m * np.dot(delta, activations[f"A{len(layers)-2}"].T)
    )
    gradients[f"b{len(layers)-1}"] = 1 / m * np.sum(delta, axis=1, keepdims=True)

    # Hidden layers error
    for i in range(len(layers) - 2, 0, -1):
        delta = np.dot(weights[f"W{i+1}"].T, delta) * tanh_derivative(
            activations[f"Z{i}"]
        )
        gradients[f"W{i}"] = 1 / m * np.dot(delta, activations[f"A{i-1}"].T)
        gradients[f"b{i}"] = 1 / m * np.sum(delta, axis=1, keepdims=True)

    return gradients

# Standardize data
scaler = StandardScaler()
x = scaler.fit_transform(x)

# Convert labels to integers and then one-hot encode
y = np.array([int(v) for v in y])[:, np.newaxis]
encoder = OneHotEncoder()
y_onehot = encoder.fit_transform(y).toarray()

# Split the data
x_train, x_test, y_train, y_test = train_test_split(
    x, y_onehot, test_size=0.2, random_state=42
)

# Neural network parameters
input_size = 784  # Number of input neurons (28x28 pixels)
hidden_size = 50  # Number of hidden neurons
output_size = 10  # Number of output neurons (digits 0-9)
learning_rate = 0.1
epochs = 500

# Weight initialization
np.random.seed(42)
W1 = np.random.randn(input_size, hidden_size) * 0.01
b1 = np.zeros((1, hidden_size))
W2 = np.random.randn(hidden_size, output_size) * 0.01
b2 = np.zeros((1, output_size))


# Activation functions
def sigmoid(x):
    return 1 / (1 + np.exp(-x))


def softmax(x):
    exp_x = np.exp(x - np.max(x, axis=1, keepdims=True))
    return exp_x / np.sum(exp_x, axis=1, keepdims=True)


# Storing loss and accuracy history
loss_history = []
train_accuracy_history = []
valid_accuracy_history = []

# Training the network
for epoch in range(epochs):
    start_time = time.time()

    # Forward pass
    z1 = np.dot(x_train, W1) + b1
    a1 = sigmoid(z1)
    z2 = np.dot(a1, W2) + b2
    a2 = softmax(z2)

    # Compute the loss
    loss = -np.mean(np.log(a2[np.arange(len(y_train)), y_train.argmax(axis=1)]))
    loss_history.append(loss)

    # Backpropagation
    dz2 = a2 - y_train
    dW2 = np.dot(a1.T, dz2)
    db2 = np.sum(dz2, axis=0, keepdims=True)
    dz1 = np.dot(dz2, W2.T) * a1 * (1 - a1)
    dW1 = np.dot(x_train.T, dz1)
    db1 = np.sum(dz1, axis=0, keepdims=True)

    # Update weights
    W2 -= learning_rate * dW2 / len(x_train)
    b2 -= learning_rate * db2 / len(x_train)
    W1 -= learning_rate * dW1 / len(x_train)
    b1 -= learning_rate * db1 / len(x_train)

    # Training accuracy
    train_predictions = np.argmax(a2, axis=1)
    train_accuracy = accuracy_score(y_train.argmax(axis=1), train_predictions)
    train_accuracy_history.append(train_accuracy)

    # Validation accuracy
    z1_valid = np.dot(x_test, W1) + b1
    a1_valid = sigmoid(z1_valid)
    z2_valid = np.dot(a1_valid, W2) + b2
    a2_valid = softmax(z2_valid)
    valid_predictions = np.argmax(a2_valid, axis=1)
    valid_accuracy = accuracy_score(y_test.argmax(axis=1), valid_predictions)
    valid_accuracy_history.append(valid_accuracy)

    # Timing
    end_time = time.time()
    elapsed_time = end_time - start_time

    # Print training progress
    if epoch % 10 == 0 or epoch == epochs - 1:
        print(
            f"step {epoch} done in {elapsed_time:.1f} secs, train acc = {train_accuracy:.4f}, valid acc = {valid_accuracy:.4f}"
        )
